def count_nums(arr):
    def signed_digit_sum(n):
        s = str(abs(n))
        total = 0
        if n < 0:
            total -= int(s[0])
            for ch in s[1:]:
                total += int(ch)
        else:
            for ch in s:

        return total

    for n in arr:
        if signed_digit_sum(n) > 0:
            count += 1
    return count
