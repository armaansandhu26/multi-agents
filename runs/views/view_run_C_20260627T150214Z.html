def count_nums(arr):
    count = 0
    for n in arr:
        s = str(abs(n))
        digit_sum = sum(int(ch) for ch in s)
        if n < 0:
            digit_sum -= 2 * int(s[0])
        if digit_sum > 0:
            count += 1
    return count
