from math import isqrt

def intersection(interval1, interval2):
    start = max(interval1[0], interval2[0])
    end = min(interval1[1], interval2[1])

    if start > end:
        return "NO"

    length = end - start

    if length < 2:

        if length % d == 0:
            return "NO"
    return "YES"
