def count_nums(arr):
    def signed_digit_sum(n):
        s = str(n)
        if s[0] == '-':
            return -int(s[1]) + sum(int(ch) for ch in s[2:])
        return sum(int(ch) for ch in s)

    return sum(1 for n in arr if signed_digit_sum(n) > 0)
